#set(comp_compile_opts ${comp_compile_opts} "$<$<COMPILE_LANGUAGE:CXX>:-std=gnu++2a>")


include($ENV{IDF_PATH}/tools/cmake/project.cmake)

if(DEFINED ENV{DISTRO})

add_compile_definitions(CELLAR)
endif()
